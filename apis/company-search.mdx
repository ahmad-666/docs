            sdk := cufinder.NewSDK("your-api-key-here")

                APIKey:     "your-api-key-here",
                BaseURL:    "https://api.cufinder.io/v2",
                Timeout:    60 * time.Second,
                MaxRetries: 3,
            })
        }

        result, err := sdk.CSE(cufinder.CseParams{
            Name:    "cufinder",
            Country: "germany",
            State:   "hamburg",
            City:    "hamburg",
        })
        if err != nil {
            log.Fatal(err)
        }
        fmt.Println(result)
